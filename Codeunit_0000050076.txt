    VAR
      DimMgt@1170000002 : Codeunit 408;
      Text014@1170000007 : TextConst 'ENU=The combination of dimensions used in %1 %2, %3, %4 is blocked. %5;RUS=Комбинация измерений, используемая в %1 %2, %3, %4, блокирована. %5';
      Text015@1170000006 : TextConst 'ENU=A dimension used in %1 %2, %3, %4 has caused an error. %5;RUS=Измерение, использованное в %1 %2, %3, %4 является причиной ошибки. %5';
      DiffAmountAndQtySignErr@1170000000 : TextConst 'ENU=Quantity and Amount must have the same sign.;RUS=Поля Количество и Сумма должы быть одного знака.';

    PROCEDURE RunCheck@2(VAR FIJnlLine@1000 : Record 50220;VAR JnlLineDim@1001 : Record 356);
    VAR
      TableID@1002 : ARRAY [10] OF Integer;
      No@1003 : ARRAY [10] OF Code[20];
    BEGIN
      WITH FIJnlLine DO BEGIN
        IF "FI No." = '' THEN
          EXIT;
        TESTFIELD("Posting Date");
        TESTFIELD("Deal Time");
        TESTFIELD("Document No.");
        TESTFIELD("External Document ID");
        TESTFIELD("FI Posting Book");
        TESTFIELD("FI Portfolio");

        IF (Quantity <> 0) AND (Amount <> 0) THEN
          IF Quantity / ABS(Quantity) <> Amount / ABS(Amount) THEN
            ERROR(DiffAmountAndQtySignErr);

        IF NOT DimMgt.CheckDimIDConsistency(JnlLineDim) THEN
          ERROR(
            Text014,
            TABLECAPTION,"Journal Template Name","Journal Batch Name","Line No.",
            DimMgt.GetDimCombErr);

        TableID[1] := DATABASE::"Financial Instrument";
        No[1] := "FI No.";
        IF NOT DimMgt.InsertObject(JnlLineDim,TableID,No) THEN
          IF "Line No." <> 0 THEN
            ERROR(
              Text015,
              TABLECAPTION,"Journal Template Name","Journal Batch Name","Line No.",
              DimMgt.GetDimValuePostingErr)
          ELSE
            ERROR(DimMgt.GetDimValuePostingErr);
      END;
    END;

    BEGIN
    {
      NC NCS-2360 ORM Created
    }
    END.
  }
