    PROCEDURE GETVendorVATInvoiceDate@1210036(VATEntry@1210000 : Record 254) : Date;
    VAR
      VendorLedgerEntry@1210001 : Record 25;
    BEGIN
      WITH VATEntry DO BEGIN
        IF Type = Type::Purchase THEN
          IF VendorLedgerEntry.GET("CV Ledg. Entry No.") THEN
            EXIT(VendorLedgerEntry."Vendor VAT Invoice Date");
      END;
    END;

    PROCEDURE InitializeCustVendInfo@1210001(VATEntry@1210000 : Record 254);
    VAR
      LocRepMgt@1210001 : Codeunit 12401;
      Vendor@1210002 : Record 23;
      Customer@1210003 : Record 18;
    BEGIN
      WITH VATEntry DO
        CASE Type OF
          Type::Purchase:
            BEGIN
              IsVendor := TRUE;
              ContractorName := LocRepMgt.GetVendorName("Bill-to/Pay-to No.");
              IF Vendor.GET("Bill-to/Pay-to No.") THEN
                VendorURL.SETPOSITION(Vendor.GETPOSITION);
            END;
          Type::Sale:
            BEGIN
              IsVendor := FALSE;
              ContractorName := LocRepMgt.GetCustName("Bill-to/Pay-to No.");
              IF Customer.GET("Bill-to/Pay-to No.") THEN
                CustomerURL.SETPOSITION(Customer.GETPOSITION);
            END;
        END;
    END;

    PROCEDURE CheckVendorAgreement@1210006(AgreementNo@1210000 : Code[20];CVLedgEntryNo@1210001 : Integer) : Boolean;
    VAR
      VendorLedgEntry@1210002 : Record 25;
    BEGIN
      IF AgreementNo = '' THEN
        EXIT(TRUE);

      IF VendorLedgEntry.GET(CVLedgEntryNo) THEN
        EXIT(VendorLedgEntry."Agreement No." = AgreementNo);

      EXIT(FALSE);
    END;

    BEGIN
    END.
