OBJECT Page 1061 Select Payment Service
{
  OBJECT-PROPERTIES
  {
    Date=25.10.16;
    Time=12:00:00;
    Version List=NAVW110.00;
  }
  PROPERTIES
  {
    CaptionML=[ENU=Select Payment Service;
               RUS=Выбор службы платежей];
    InsertAllowed=No;
    DeleteAllowed=No;
    SourceTable=Table1060;
    DataCaptionExpr='';
    PageType=StandardDialog;
    OnQueryClosePage=VAR
                       TempPaymentServiceSetup@1000 : TEMPORARY Record 1060;
                     BEGIN
                       IF CloseAction IN [ACTION::Cancel,ACTION::LookupCancel] THEN
                         EXIT;

                       IF DeselectedValue THEN
                         EXIT(TRUE);

                       TempPaymentServiceSetup.COPY(Rec,TRUE);
                       TempPaymentServiceSetup.SETRANGE(Available,TRUE);
                       IF NOT TempPaymentServiceSetup.FINDFIRST THEN
                         EXIT(CONFIRM(NoPaymentServicesSelectedQst));
                     END;

  }
  CONTROLS
  {
    { 1   ;0   ;Container ;
                ContainerType=ContentArea }

    { 3   ;1   ;Group     ;
                GroupType=Repeater }

    { 4   ;2   ;Field     ;
                ToolTipML=[ENU=Specifies that the icon and link to the payment service will be inserted on the outgoing sales document.;
                           RUS=Указывает, что в исходящий документ продажи будут вставлены значок и ссылка на службу платежей.];
                ApplicationArea=#Basic,#Suite;
                SourceExpr=Available;
                OnValidate=BEGIN
                             IF NOT Available THEN
                               DeselectedValue := TRUE;
                           END;
                            }

    { 5   ;2   ;Field     ;
                ToolTipML=[ENU=Specifies the name of the payment service.;
                           RUS=Определяет название службы платежей.];
                ApplicationArea=#Basic,#Suite;
                SourceExpr=Name;
                Editable=FALSE }

    { 2   ;2   ;Field     ;
                ToolTipML=[ENU=Specifies the description of the payment service.;
                           RUS=Определяет описание службы платежей.];
                ApplicationArea=#Basic,#Suite;
                SourceExpr=Description;
                Editable=FALSE }

    { 6   ;1   ;Field     ;
                Name=SetupPaymentServices;
                CaptionML=[ENU=SetupPaymentServices;
                           RUS=Настройка службы платежей];
                ApplicationArea=#Basic,#Suite;
                SourceExpr=SetupPaymentServicesLbl;
                Editable=FALSE;
                OnDrillDown=BEGIN
                              CurrPage.CLOSE;
                              PAGE.RUN(PAGE::"Payment Services");
                            END;

                ShowCaption=No }

  }
  CODE
  {
    VAR
      DeselectedValue@1000 : Boolean;
      NoPaymentServicesSelectedQst@1001 : TextConst 'ENU=To enable the payment service for the document, you must select the Available check box.\\Are you sure you want to exit?;RUS=Чтобы включить службу платежей для документа, необходимо установить флажок "Доступно".\\Вы действительно хотите выйти?';
      SetupPaymentServicesLbl@1002 : TextConst 'ENU=Set Up Payment Services;RUS=Настройка службы платежей';

    BEGIN
    END.
  }
