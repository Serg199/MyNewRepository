    PROCEDURE GetDimCodeFromID2@1101495002(DimCode@1101495000 : Code[20];VAR Code1@1101495001 : Code[20];ID1@1101495002 : Code[40];VAR Code2@1101495005 : Code[20];ID2@1101495004 : Code[40]);
    VAR
      DimValue@1101495003 : Record 349;
      ExitCode1@1101495006 : Code[20];
      ExitCode2@1101495007 : Code[20];
      isDifferent@1101495008 : Boolean;
    BEGIN
      //NC S547.1 > PCH
      IF (DimCode = '') OR (ID1 = '') THEN
        ExitCode1 := Code1;
      IF (DimCode = '') OR (ID2 = '') THEN
        ExitCode2 := Code2;
      IF Code1 = ID1 THEN
        ExitCode1 := Code1;
      IF Code2 = ID2 THEN
        ExitCode2 := Code2;
      IF (ExitCode1 <> '') AND (ExitCode2 <> '') THEN BEGIN
        Code1 := ExitCode1;
        Code2 := ExitCode2;
        EXIT;
      END;
      DimValue.SETRANGE("Dimension Code",DimCode);
      isDifferent := FALSE;
      CASE TRUE OF
        (ID1 = ID2) AND (ID1 <> ''): DimValue.SETRANGE("Rec ID",ID1);
        (ExitCode1 = '') AND (ExitCode2 = '') AND (ID1 <> '') AND (ID2 <> '') :
          BEGIN
            isDifferent := TRUE;
            DimValue.SETFILTER("Rec ID",'%1|%2',ID1,ID2);
          END;
        (ExitCode1 = '') AND (ID1 <> '') : DimValue.SETRANGE("Rec ID",ID1);
        (ExitCode2 = '') AND (ID2 <> '') : DimValue.SETRANGE("Rec ID",ID2);
        ELSE BEGIN
          IF ExitCode1 <> '' THEN
            Code1 := ExitCode1;
          IF ExitCode2 <> '' THEN
            Code2 := ExitCode2;
          EXIT;
        END;
      END;
      IF DimValue.ISEMPTY THEN BEGIN
        IF ExitCode1 <> '' THEN
          Code1 := ExitCode1;
        IF ExitCode2 <> '' THEN
          Code2 := ExitCode2;
        EXIT;
      END;

      IF isDifferent THEN BEGIN  // Значения измерений разные
        IF DimValue.FINDSET THEN REPEAT
          IF (ExitCode1 = '') AND (DimValue."Rec ID" = ID1) THEN
            ExitCode1 := DimValue.Code;
          IF (ExitCode2 = '') AND (DimValue."Rec ID" = ID2) THEN
            ExitCode2 := DimValue.Code;
        UNTIL DimValue.NEXT = 0;
      END ELSE BEGIN
        IF DimValue.FINDFIRST THEN BEGIN
          IF (ExitCode1 = '') AND (DimValue."Rec ID" = ID1) THEN
            ExitCode1 := DimValue.Code;
          IF (ExitCode2 = '') AND (DimValue."Rec ID" = ID2) THEN
            ExitCode2 := DimValue.Code;
        END;
      END;

      IF ExitCode1 <> '' THEN
        Code1 := ExitCode1;
      IF ExitCode2 <> '' THEN
        Code2 := ExitCode2;
      //NC S547.1 < PCH
    END;
