OBJECT Page 9810 Set Password
{
  OBJECT-PROPERTIES
  {
    Date=25.10.16;
    Time=12:00:00;
    Version List=NAVW110.00;
  }
  PROPERTIES
  {
    CaptionML=[ENU=Set Password;
               RUS=Установить пароль];
    SourceTable=Table2000000120;
    DataCaptionExpr="Full Name";
    PageType=StandardDialog;
    OnQueryClosePage=BEGIN
                       IF CloseAction = ACTION::OK THEN BEGIN
                         IF SetPassword <> ConfirmPassword THEN
                           ERROR(Text001);
                         IF IdentityManagement.ValidatePasswordStrength(SetPassword) THEN BEGIN
                           SETUSERPASSWORD("User Security ID",SetPassword);
                         END ELSE
                           ERROR(Text002);
                       END
                     END;

  }
  CONTROLS
  {
    { 1   ;0   ;Container ;
                ContainerType=ContentArea }

    { 2   ;1   ;Field     ;
                Name=<SetPassword>;
                ExtendedDatatype=Masked;
                CaptionML=[ENU=Password;
                           RUS=Пароль];
                ApplicationArea=#Basic,#Suite;
                SourceExpr=SetPassword;
                OnValidate=BEGIN
                             IF NOT IdentityManagement.ValidatePasswordStrength(SetPassword) THEN
                               ERROR(Text002);
                           END;
                            }

    { 3   ;1   ;Field     ;
                Name=<ConfirmPassword>;
                ExtendedDatatype=Masked;
                CaptionML=[ENU=Confirm Password;
                           RUS=Подтвердить пароль];
                ToolTipML=[ENU=Specifies the password repeated.;
                           RUS=Определяет повтор пароля.];
                ApplicationArea=#Basic,#Suite;
                SourceExpr=ConfirmPassword;
                OnValidate=BEGIN
                             IF SetPassword <> ConfirmPassword THEN
                               ERROR(Text001);
                           END;
                            }

  }
  CODE
  {
    VAR
      Text001@1003 : TextConst 'ENU=The passwords that you entered do not match.;RUS=Введенные пароли не совпадают.';
      Text002@1004 : TextConst 'ENU=The password that you entered does not meet the minimum requirements. It must be at least 8 characters long and contain at least one uppercase letter, one lowercase letter, and one number.;RUS=Введенный пароль не удовлетворяет минимальным требованиям. Он должен быть не короче 8 символов и должен содержать как минимум одну прописную букву, одну строчную букву и одну цифру.';
      IdentityManagement@1002 : Codeunit 9801;
      SetPassword@1000 : Text[250] INDATASET;
      ConfirmPassword@1001 : Text[250] INDATASET;

    BEGIN
    END.
  }
}
