    LOCAL PROCEDURE GetLaterLabel@38() : Text;
    BEGIN
      CASE TRUE OF
        UserCanContinue AND NOT UserCanUpdate:
          EXIT(ContinueLbl);
        ELSE
          EXIT(UpgradeLaterLbl);
      END;
    END;

    END.
