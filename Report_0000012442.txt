    PROCEDURE GetClosedAmount@1210024(ApplicationAmount@1210003 : Decimal;AppliedAmount@1210002 : Decimal) : Decimal;
    VAR
      Amount@1210000 : Decimal;
    BEGIN
      IF ABS(ApplicationAmount) > ABS(AppliedAmount) THEN
        Amount := ABS(AppliedAmount)
      ELSE
        Amount := ABS(ApplicationAmount);

      IF ApplicationAmount > 0 THEN
        EXIT(Amount)
      ELSE
        EXIT(-Amount);
    END;
