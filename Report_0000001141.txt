OBJECT Report 1141 Delete Old Cost Entries
{
  OBJECT-PROPERTIES
  {
    Date=18.12.13;
    Time=12:00:00;
    Version List=NAVW17.10;
  }
  PROPERTIES
  {
    CaptionML=[ENU=Delete Old Cost Entries;
               RUS=Удалить старые записи затрат];
    ProcessingOnly=Yes;
  }
  DATASET
  {
  }
  REQUESTPAGE
  {
    PROPERTIES
    {
    }
    CONTROLS
    {
      { 1   ;    ;Container ;
                  ContainerType=ContentArea }

      { 2   ;1   ;Field     ;
                  Name=YearEndingDate;
                  CaptionML=[ENU=Year Ending Date;
                             RUS=Дата окончания года];
                  SourceExpr=YearEndDate;
                  OnValidate=BEGIN
                               IF YearEndDate <> CALCDATE('<CY>',YearEndDate) THEN
                                 ERROR(Text001,YearEndDate);

                               IF WORKDATE - YearEndDate < 365 THEN
                                 ERROR(Text002,YearEndDate);

                               IF NOT CONFIRM(Text003,FALSE,YearEndDate) THEN
                                 EXIT;

                               CostEntry.SETCURRENTKEY("Cost Type No.","Posting Date");
                               CostEntry.SETRANGE("Posting Date",0D,YearEndDate);
                               IF NOT CostEntry.ISEMPTY THEN BEGIN
                                 CostEntry.DELETEALL;
                                 MESSAGE(Text004,YearEndDate);
                               END ELSE
                                 ERROR(Text005,YearEndDate);
                             END;
                              }

    }
  }
  LABELS
  {
  }
  CODE
  {
    VAR
      CostEntry@1006 : Record 1104;
      YearEndDate@1000 : Date;
      Text001@1005 : TextConst 'ENU=%1 is not at year''s end.;RUS=%1 не приходится на конец года.';
      Text002@1004 : TextConst 'ENU=The selected year ending date %1 must be older than last year.;RUS=Дата завершения для выбранного года, %1, должна предшествовать дате для последнего года.';
      Text003@1003 : TextConst 'ENU=Are you sure you want to delete all cost entries up to and including %1?;RUS=Действительно удалить все операции затрат до %1 включительно?';
      Text004@1002 : TextConst 'ENU=All cost entries up to and including %1 deleted.;RUS=Все операции затрат до %1 включительно удалены.';
      Text005@1001 : TextConst 'ENU=No cost entries were found before %1.;RUS=До %1 не найдена ни одна операция затрат.';

    BEGIN
    END.
  }
  RDLDATA
  {
  }
}
