OBJECT Codeunit 83 Sales-Quote to Order (Yes/No)
{
  OBJECT-PROPERTIES
  {
    Date=25.10.16;
    Time=12:00:00;
    Version List=NAVW110.00;
  }
  PROPERTIES
  {
    TableNo=36;
    OnRun=VAR
            OfficeMgt@1000 : Codeunit 1630;
          BEGIN
            TESTFIELD("Document Type","Document Type"::Quote);
            IF GUIALLOWED THEN
              IF NOT CONFIRM(ConfirmConvertToOrderQst,FALSE) THEN
                EXIT;

            IF CheckCustomerCreated(TRUE) THEN
              GET("Document Type"::Quote,"No.")
            ELSE
              EXIT;

            SalesQuoteToOrder.RUN(Rec);
            SalesQuoteToOrder.GetSalesOrderHeader(SalesHeader2);
            COMMIT;

            IF GUIALLOWED THEN
              IF OfficeMgt.AttachAvailable THEN
                PAGE.RUN(PAGE::"Sales Order",SalesHeader2)
              ELSE
                IF CONFIRM(STRSUBSTNO(OpenNewInvoiceQst,SalesHeader2."No."),TRUE) THEN
                  PAGE.RUN(PAGE::"Sales Order",SalesHeader2);
          END;

  }
  CODE
  {
    VAR
      ConfirmConvertToOrderQst@1000 : TextConst 'ENU=Do you want to convert the quote to an order?;RUS=Преобразовать предложение в заказ?';
      OpenNewInvoiceQst@1004 : TextConst '@@@="%1 = No. of the new sales order document.";ENU=The quote has been converted to order %1. Do you want to open the new order?;RUS=Предложение преобразовано в заказ %1. Открыть созданный заказ?';
      SalesHeader2@1002 : Record 36;
      SalesQuoteToOrder@1003 : Codeunit 86;

    BEGIN
    END.
  }
}
