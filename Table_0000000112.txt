    PROCEDURE FindShipments@1210000(SalesInvHeader@1210003 : Record 112);
    VAR
      ValueEntry@1210007 : Record 5802;
      ItemLedgEntry@1210006 : Record 32;
      SalesShptHeader@1210005 : Record 110;
      SalesShptLine@1210004 : Record 111;
      SalesInvLine@1210002 : Record 113;
      DocNoFilter@1210001 : Text[1024];
      I@1210000 : Integer;
      Text12400@1210008 : TextConst 'ENU=Length of the Document No. filter should not exceed 1024.;RUS=Длина фильтра по номеру документа не должна превышать 1024.';
    BEGIN
      DocNoFilter := '';
      I := 0;

      SalesInvLine.RESET;
      SalesInvLine.SETRANGE("Document No.",SalesInvHeader."No.");
      SalesInvLine.SETRANGE(Type,SalesInvLine.Type::Item);
      SalesInvLine.SETFILTER(Quantity,'>%1',0);
      IF SalesInvLine.FIND('-') THEN
        REPEAT
          ValueEntry.RESET;
          ValueEntry.SETCURRENTKEY("Item No.","Posting Date","Document No.");
          ValueEntry.SETRANGE("Document No.",SalesInvHeader."No.");
          ValueEntry.SETRANGE("Posting Date",SalesInvHeader."Posting Date");
          ValueEntry.SETRANGE("Item No.",SalesInvLine."No.");
          IF ValueEntry.FIND('-') THEN
            REPEAT
              ItemLedgEntry.GET(ValueEntry."Item Ledger Entry No.");
              SalesShptHeader.GET(ItemLedgEntry."Document No.");
              I := I + 1;
              IF I = 1 THEN
                DocNoFilter := SalesShptHeader."No."
              ELSE BEGIN
                IF STRPOS('|'+DocNoFilter+'|','|'+SalesShptHeader."No."+'|') = 0 THEN
                  IF (STRLEN(DocNoFilter) + STRLEN(SalesShptHeader."No.")) < MAXSTRLEN(DocNoFilter) THEN
                    DocNoFilter := DocNoFilter + '|' + SalesShptHeader."No."
                  ELSE
                    ERROR(Text12400);
              END;
            UNTIL ValueEntry.NEXT = 0;
        UNTIL SalesInvLine.NEXT = 0;

      IF DocNoFilter = '' THEN
        DocNoFilter := '.';
      SalesShptHeader.RESET;
      SalesShptHeader.SETFILTER("No.",DocNoFilter);
      PAGE.RUN(PAGE::"Posted Sales Shipments",SalesShptHeader);
    END;
