                       Style := FALSE;
                       PublisherOrStatus := Publisher;

                      Scope=Repeater;

                               END;

                      Image=RefreshLines;
                      PromotedCategory=Category5;
                      OnAction=BEGIN
                                 ActionsEnabled := FALSE;

                      PromotedIsBig=Yes;

                                   CurrPage.UPDATE;
                               END;
                                }
      { 15      ;1   ;ActionGroup;
                      Name=Manage;
                      CaptionML=[ENU=Manage;
                                 RUS=Управление] }
      { 10      ;2   ;Action    ;
                      Name=View;
                      ShortCutKey=Return;
                      CaptionML=[ENU=View;
                                 RUS=Просмотр];
                      ToolTipML=[ENU=View extension details.;

                      Enabled=ActionsEnabled;
                      Image=View;
                      PromotedCategory=Category4;
