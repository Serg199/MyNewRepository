OBJECT Codeunit 1108 CA Jnl.-Post
{
  OBJECT-PROPERTIES
  {
    Date=25.10.16;
    Time=12:00:00;
    Version List=NAVW110.00;
  }
  PROPERTIES
  {
    TableNo=1101;
    OnRun=BEGIN
            CostJnlLine.COPY(Rec);
            Code;
            Rec := CostJnlLine;
          END;

  }
  CODE
  {
    VAR
      CostJnlLine@1000 : Record 1101;
      Text001@1001 : TextConst 'ENU=Do you want to post the journal lines?;RUS=Учесть строки журнала?';
      Text002@1002 : TextConst 'ENU=There is nothing to post.;RUS=Здесь нечего учитывать.';
      Text003@1003 : TextConst 'ENU=The journal lines were successfully posted.;RUS=Строки журнала успешно учтены.';
      Text004@1004 : TextConst 'ENU=The journal lines were successfully posted. You are now in the %1 journal.;RUS=Строки журнала успешно учтены. Вы находитесь в журнале %1.';

    LOCAL PROCEDURE Code@1();
    VAR
      TempJnlBatchName@1000 : Code[10];
    BEGIN
      WITH CostJnlLine DO BEGIN
        IF NOT CONFIRM(Text001) THEN
          EXIT;

        TempJnlBatchName := "Journal Batch Name";
        CODEUNIT.RUN(CODEUNIT::"CA Jnl.-Post Batch",CostJnlLine);

        IF "Line No." = 0 THEN
          MESSAGE(Text002)
        ELSE
          IF TempJnlBatchName = "Journal Batch Name" THEN
            MESSAGE(Text003)
          ELSE
            MESSAGE(Text004,"Journal Batch Name");

        IF NOT FIND('=><') OR (TempJnlBatchName <> "Journal Batch Name") THEN BEGIN
          RESET;
          FILTERGROUP(2);
          SETRANGE("Journal Template Name","Journal Template Name");
          SETRANGE("Journal Batch Name","Journal Batch Name");
          FILTERGROUP(0);
          "Line No." := 1;
        END;
      END;
    END;

    BEGIN
    END.
  }
}
