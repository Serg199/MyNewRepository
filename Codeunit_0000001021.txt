OBJECT Codeunit 1021 Job Jnl.-Post
{
  OBJECT-PROPERTIES
  {
    Date=25.10.16;
    Time=12:00:00;
    Version List=NAVW110.00;
  }
  PROPERTIES
  {
    TableNo=210;
    OnRun=BEGIN
            JobJnlLine.COPY(Rec);
            Code;
            COPY(JobJnlLine);
          END;

  }
  CODE
  {
    VAR
      Text000@1000 : TextConst 'ENU=cannot be filtered when posting recurring journals.;RUS=не может фильтроваться при учете типовых журналов.';
      Text001@1001 : TextConst 'ENU=Do you want to post the journal lines?;RUS=Учесть строки журнала?';
      Text002@1002 : TextConst 'ENU=There is nothing to post.;RUS=Здесь нечего учитывать.';
      Text003@1003 : TextConst 'ENU=The journal lines were successfully posted.;RUS=Строки журнала успешно учтены.';
      Text004@1004 : TextConst 'ENU="The journal lines were successfully posted. ";RUS="Строки журнала успешно учтены. "';
      Text005@1005 : TextConst 'ENU=You are now in the %1 journal.;RUS=Вы сейчас в журнале %1.';
      JobJnlTemplate@1006 : Record 209;
      JobJnlLine@1007 : Record 210;
      TempJnlBatchName@1009 : Code[10];

    LOCAL PROCEDURE Code@1();
    BEGIN
      WITH JobJnlLine DO BEGIN
        JobJnlTemplate.GET("Journal Template Name");
        JobJnlTemplate.TESTFIELD("Force Posting Report",FALSE);
        IF JobJnlTemplate.Recurring AND (GETFILTER("Posting Date") <> '') THEN
          FIELDERROR("Posting Date",Text000);

        IF NOT CONFIRM(Text001) THEN
          EXIT;

        TempJnlBatchName := "Journal Batch Name";

        CODEUNIT.RUN(CODEUNIT::"Job Jnl.-Post Batch",JobJnlLine);

        IF "Line No." = 0 THEN
          MESSAGE(Text002)
        ELSE
          IF TempJnlBatchName = "Journal Batch Name" THEN
            MESSAGE(Text003)
          ELSE
            MESSAGE(
              Text004 +
              Text005,
              "Journal Batch Name");

        IF NOT FIND('=><') OR (TempJnlBatchName <> "Journal Batch Name") THEN BEGIN
          RESET;
          FILTERGROUP(2);
          SETRANGE("Journal Template Name","Journal Template Name");
          SETRANGE("Journal Batch Name","Journal Batch Name");
          FILTERGROUP(0);
          "Line No." := 1;
        END;
      END;
    END;

    BEGIN
    END.
  }
}
