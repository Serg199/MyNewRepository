    PROCEDURE SetBody@1(BodyText@1000 : Text);
    VAR
      OutStream@1001 : OutStream;
    BEGIN
      CALCFIELDS(Body);

    PROCEDURE GetBody@2() BodyText : Text;
    VAR
      InStream@1000 : InStream;
    BEGIN

    END;
